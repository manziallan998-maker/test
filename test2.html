DATABASE LEVEL PRODUCTION DECLARATIONS (POSTGRESQL) 
-- =====================================================================
-- PRODUCTION-GRADE POSTGRESQL LAYER SCHEMA DESIGN — ADMISSIONIQ
-- PROJECT REF HOOK: SECURE MULTI-TENANT ISOLATED SAAS CORE
-- =====================================================================

CREATE EXTENSION IF NOT EXISTS "uuid-ossp";
CREATE EXTENSION IF NOT EXISTS "pg_trgm";

-- USER PROFILE RECORD AND PRIVILEGE SYSTEM MAP
CREATE TABLE IF NOT EXISTS public.profiles (
    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    full_name TEXT NOT NULL,
    email TEXT UNIQUE NOT NULL,
    country TEXT NOT NULL,
    intended_degree TEXT,
    preferred_countries TEXT,
    plan TEXT NOT NULL DEFAULT 'free' CHECK (plan IN ('free', 'pro', 'enterprise')),
    created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- INSTITUTIONAL GLOBAL PARSING ENGINE DATA
CREATE TABLE IF NOT EXISTS public.universities (
    id BIGSERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    country TEXT NOT NULL,
    ranking INTEGER,
    tuition_usd NUMERIC(12, 2) NOT NULL,
    requirements TEXT,
    deadline DATE,
    degree_types TEXT[] NOT NULL,
    is_active BOOLEAN DEFAULT true
);

-- SCHOLARSHIP FUNDING ENTITY RECORD
CREATE TABLE IF NOT EXISTS public.scholarships (
    id BIGSERIAL PRIMARY KEY,
    title TEXT NOT NULL,
    provider TEXT NOT NULL,
    amount_usd NUMERIC(12, 2),
    funding_type TEXT CHECK (funding_type IN ('Full', 'Partial', 'Living Allowance', 'Tuition Only')),
    deadline DATE,
    is_active BOOLEAN DEFAULT true
);

-- MASTER PIPELINE COMPLIANCE AGGREGATION TRACKER
CREATE TABLE IF NOT EXISTS public.applications (
    id BIGSERIAL PRIMARY KEY,
    user_id UUID NOT NULL REFERENCES public.profiles(id) ON DELETE CASCADE,
    university_name TEXT NOT NULL,
    program TEXT NOT NULL,
    status TEXT NOT NULL DEFAULT 'researching' CHECK (status IN ('researching','draft','in_progress','submitted','accepted','rejected')),
    created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- SUPABASE STORAGE METADATA MAPS
CREATE TABLE IF NOT EXISTS public.documents (
    id BIGSERIAL PRIMARY KEY,
    user_id UUID NOT NULL REFERENCES public.profiles(id) ON DELETE CASCADE,
    doc_type TEXT NOT NULL,
    doc_name TEXT NOT NULL,
    file_url TEXT NOT NULL,
    created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- ENABLE AND DECLARE STRICT ROW-LEVEL SECURITY POLICIES
ALTER TABLE public.profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.applications ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Profile Isolation Vector" ON public.profiles 
    FOR ALL USING (auth.uid() = id);

CREATE POLICY "Application Multi-Tenant Isolation" ON public.applications 
    FOR ALL USING (auth.uid() = user_id);
